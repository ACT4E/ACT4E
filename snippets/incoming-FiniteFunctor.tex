class FiniteFunctor(ABC):
    @abstractmethod
    def source(self) -> FiniteCategory:
        ...
    @abstractmethod
    def target(self) -> FiniteCategory:
        ...
    @abstractmethod
    def f_ob(self, ob: Object) -> Object:
        """ Effect on objects """
    @abstractmethod
    def f_mor(self, m: Morphism) -> Morphism:
        """ Effect on morphisms """
