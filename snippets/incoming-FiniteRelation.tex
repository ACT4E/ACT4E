class FiniteRelation(ABC):
    @abstractmethod
    def source(self) -> FiniteSet:
        """ Returns a finite set"""
    @abstractmethod
    def target(self) -> FiniteSet:
        """ Returns a finite set"""
    @abstractmethod
    def holds(self, a: Element, b: Element) -> bool:
        """ Returns true if the two elements are related """
