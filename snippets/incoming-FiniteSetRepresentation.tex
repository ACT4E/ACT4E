class FiniteSetRepresentation(ABC):
    @abstractmethod
    def load(self, yaml_data: str) -> FiniteSet:
        """ Load a finite set from given YAML data.
            Throw InvalidFormat if the format is incorrect.
        """
    @abstractmethod
    def save(self, f: FiniteSet) -> str:
        """ Save to a string. """
