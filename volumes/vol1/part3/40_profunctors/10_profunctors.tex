\section{Profunctors}

\begin{ctdefinition}[Profunctor]
  \label{def:profunctor}
  Given two categories \CatC and \CatD, a \emph{\iindex{profunctor}} from \CatC to \CatD is a functor of the form
  \begin{equation}
    H\colon \CatC\op \times \CatD \to \Set.
  \end{equation}
\end{ctdefinition}
