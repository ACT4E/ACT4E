    \begin{definition}[Uncertain Design Problem]
        \label{def:unc_dp}
        An Uncertain DP (UDP)~$u$ is a pair of DPs~$\tup{\udpL u,\udpU u}$
        such that~$\udpL u \dpleq \udpU u$.
    \end{definition}

    \begin{lemma}
        \label{lem:well-formed-series-bis}$\pars{\udpL f\mthen \udpL g}\dpleq\pars{\udpU f\mthen\udpU g}$.
    \end{lemma}
    \begin{proof}
        We already proved that~$\mthen$ is monotone.
        Then note that
        \begin{equation}
            \label{eq:series_unc_mon}
            \pars{\udpL f\mthen\udpL g}\dpleq\pars{\udpL f\mthen\udpU g}\dpleq\pars{\udpU f\mthen\udpU g}.
        \end{equation}
    \end{proof}
    \begin{lemma}
        \label{lem:well-formed-par-bis}$\pars{\udpL x\mtimescat\udpL y}\dpleq\pars{\udpU x\mtimescat\udpU y}$.
    \end{lemma}
    \begin{proof}
        We already proved that~$\mtimescat$ is monotone in each argument.
        Then note that
        \begin{equation}
            \label{eq:parallel_unc_mon}
            \pars{\udpL f\mtimescat\udpL g}\dpleq\pars{\udpL f\mtimescat\udpU g}\dpleq\pars{\udpU f\mtimescat\udpU g}.
        \end{equation}
    \end{proof}
