\begin{ctdefinition}[Category of real matrices]
    \SYNDEF{category of real matrices}
    \label{def:cat-of-real-matrices}
    The category $\CatStyle{Mat}_\reals$ of real matrices is:
    \begin{enumerate}
        \item \emph{Objects}: $\styleobj{\reals^n}$, for $n \setin \natnumbers$.
        \item \emph{Morphisms}: for any $n, m \setin \natnumbers$, $\HomSet{\Set}{\styleobj{\reals^m}}{\styleobj{\reals^n}}$ is the set  of $n \times m$ real matrices.
        \item \emph{Composition}: matrix multiplication.
        \item \emph{Identity morphisms}: identity matrices.
    \end{enumerate}
\end{ctdefinition}

\begin{ctdefinition}[Category of real vector spaces]
    \SYNDEF{category of real vector spaces}
    \label{def:cat-of-real-vector-spaces}
    The category $\CatStyle{Vect}_\reals$ of real vector spaces is:
    \begin{enumerate}
        \item \emph{Objects}: all real vector spaces.
        \item \emph{Morphisms}: $\HomSet{\Set}{\Obja}{\Objb}$ is the set of real linear maps $\Obja \mto \Objb$.
        \item \emph{Composition}: the usual composition of linear maps.
        \item \emph{Identity morphisms}: for any real vector space $\Obja = \tup{\stylesets{X}, +, \cdot}$, the identity morphism $\catid_\Obja$ is given by the identity function $\stylesets{X} \to \stylesets{X}$.
    \end{enumerate}
\end{ctdefinition}
