\section[Categories of semigroups, monoids, groups]{Categories of semigroups, monoids, groups}

\begin{ctdefinition}[Category of semigroups]
    \SYNDEF{category of semigroups and morphisms}
    \label{def:SGrp}
    The category $\SGrp$ of \SY{semigroups} is defined by:
    \begin{enumerate}
        \item \emph{Objects}: all \SY{semigroups}.
        \item \emph{Morphisms}: for \SY{semigroups} $\Obja$ and $\Objb$, %$\Obja = \tup{\sgrpAset,{{ \mthenof \Obja}}}$ and $\Objb = \tup{\sgrpBset,{{ \mthenof \Objb}}}$,
              the hom-set $\HomSet{\SGrp}{\Obja}{\Objb}$ is the set of all \SY{semigroup morphisms} from $\Obja$ to~$\Objb$.
        \item \emph{Composition}: composition of \SY{semigroup morphisms} (composition of the underlying functions).
        \item \emph{Identity morphisms}: for a \SY{semigroup} $\Obja = \tup{\sgrpAset, {{\mthenof \Obja}}}$, its \SY{identity morphism} $\catidat\Obja$ is given by the \SY{identity function} $\sgrpAset \to \sgrpAset$.
    \end{enumerate}
\end{ctdefinition}

\begin{gradedexercise}[\exname{CategorySemigroups}]
    Check explicitly that the above definition does indeed define a category.
    Is the composition of composable morphisms again a morphism? Does the associative law hold? Are the conditions for identity morphisms satisfied?
\end{gradedexercise}

\solutionof{CategorySemigroups}

\begin{remark}
    An isomorphism in the category $\SGrp$ of semigroups is the same thing as an isomorphism of semigroups.
\end{remark}

\solutionof{CategorySemigroups}

\begin{ctdefinition}[Category of monoids]
    \SYNDEF{category of monoids and morphisms}
    \label{def:Mon}
    The category \Mon of \SY{monoids} is defined by:
    \begin{enumerate}
        \item \emph{Objects}: all \SY{monoids}.
        \item \emph{Morphisms}: for \SY{monoids} $\Obja$ and $\Objb$, %$\Obja = \tup{\monoidAset, {{\mthenof \Obja}}, \idmon_\Obja}$ and $\Objb = \tup{\monoidBset, {{\mthenof \Objb}}, \idmon_\Objb}$,
              the hom-set $\HomSet{\Mon}{\Obja}{\Objb}$ is the set of all \SY{monoid morphisms} from $\Obja$ to~$\Objb$.
        \item \emph{Composition}: composition of \SY{monoid morphisms}.
        \item \emph{Identity morphisms}: for a \SY{monoid} $\Obja = \tup{\monoidAset, {{\mthen}}, \idmon}$, its \SY{identity morphism} $\catidat\Obja$ is given by the \SY{identity function} $\monoidAset \to \monoidAset$.
    \end{enumerate}
\end{ctdefinition}

\begin{ctdefinition}[Category of groups]
    \label{def:Grp}
    \SYNDEF{category of groups and morphisms}
    The category \Grp of \SY{groups} is defined by:
    \begin{enumerate}
        \item \emph{Objects}: all \SY{groups}.
        \item \emph{Morphisms}: for \SY{groups} $\Obja$, $\Objb$, %$\Obja = \tup{\grpAset, {{\mthenof \Obja}}, \idmon_\Obja}$ and $\Objb = \tup{\grpBset, {{\mthenof \Objb}}, \idmon_\Objb}$,
              the hom-set $\HomSet{\Grp}{\Obja}{\Objb}$ is the set of all group morphisms from $\Obja$ to~$\Objb$.
        \item \emph{Composition}: composition of \SY{group morphisms}.
        \item \emph{Identity morphisms}: for a \SY{group} $\Obja$, its \SY{identity morphism} $\catidat\Obja$ is given by the \SY{identity function} $\grpAset \to \grpAset$.
    \end{enumerate}
\end{ctdefinition}
