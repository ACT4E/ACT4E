\section[Categories of semigroups, monoids, groups]{Categories of semigroups, monoids, groups}
\label{sec:cats-of-semigroups-monoids-groups}

\todojira{617}{\alphubel: @JL: move relevant material here}

\begin{ctdefinition}[Category of semigroups]
    \SYNDEF{category of semigroups and morphisms}
    \label{def:SGrp}
    The category $\SGrp$ of \SY{semigroups} is:
    \begin{enumerate}
        \item \emph{Objects}: all \SY{semigroups}.
        \item \emph{Morphisms}: for \SY{semigroups} $\Obja = \tup{\sgrpAset,{ \mthenof \Obja}}$ and $\Objb = \tup{\sgrpBset,{ \mthenof \Objb}}$, the hom-\-set $\HomSet{\SGrp}{\Obja}{\Objb}$ is the set of all \SY{semigroup morphisms} from $\Obja$ to~$\Objb$.
        \item \emph{Composition}: composition of \SY{semigroup morphisms} (composition of the underlying functions).
        \item \emph{Identity morphisms}: for a \SY{semigroup} $\Obja = \tup{\sgrpAset, {\mthenof \Obja}}$, its \SY{identity morphism} $\catid_\Obja$ is given by the \SY{identity function} $\sgrpAset \to \sgrpAset$.
    \end{enumerate}
\end{ctdefinition}

\begin{ctdefinition}[Category of monoids]
    \SYNDEF{category of monoids and morphisms}
    \label{def:Mon}
    The category \Mon of \SY{monoids} is:
    \begin{enumerate}
        \item \emph{Objects}: all \SY{monoids}.
        \item \emph{Morphisms}: for \SY{monoids} $\Obja = \tup{\monoidAset, {{\mthenof \Obja}}, \idmon_\Obja}$ and $\Objb = \tup{\monoidBset, {{\mthenof \Objb}}, \idmon_\Objb}$, the hom-set $\HomSet{\Mon}{\Obja}{\Objb}$ is the set of all \SY{monoid morphisms} from $\Obja$ to~$\Objb$.
        \item \emph{Composition}: composition of \SY{monoid morphisms}.
        \item \emph{Identity morphisms}: for a \SY{monoid} $\Obja = \tup{\monoidAset, {\mthen}, \idmon}$, its \SY{identity morphism} $\catidat\Obja$ is given by the \SY{identity function} $\monoidAset \to \monoidAset$.
    \end{enumerate}
\end{ctdefinition}

\begin{ctdefinition}[Category of groups]
    \label{def:Grp}
    \SYNDEF{category of groups and morphisms}
    The category \Grp of \SY{groups} is:
    \begin{enumerate}
        \item \emph{Objects}: all \SY{groups}.
        \item \emph{Morphisms}: for \SY{groups} $\Obja = \tup{\grpAset, {\mthenof \Obja}, \idmon_\Obja}$ and $\Objb = \tup{\grpBset, {\mthenof \Objb}, \idmon_\Objb}$,  the hom-set $\HomSet{\Grp}{\Obja}{\Objb}$ is the set of all group morphisms from $\Obja$ to~$\Objb$.
        \item \emph{Composition}: composition of \SY{group morphisms}.
        \item \emph{Identity morphisms}: for a \SY{group} $\Obja = \tup{\grpAset, {\mthen}, \idmon}$, its \SY{identity morphism} $\catid_\Obja$ is given by the \SY{identity function} $\grpAset \to \grpAset$.
    \end{enumerate}
\end{ctdefinition}
