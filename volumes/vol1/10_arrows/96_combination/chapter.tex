\chaptersecond{Combination}{chapter-edelweiss}{chap:combination}{
    This chapter discusses products and coproducts, two canonical ways of combining objects in a category.
    In the category \Set, they are \SY{cartesian product} and disjoint union.
    In a \SY{lattice}, they are the \SY{meet} and \SY{join}.
}
