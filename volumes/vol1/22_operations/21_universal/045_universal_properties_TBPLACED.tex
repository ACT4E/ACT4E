\begin{ctdefinition}[Initial and terminal object]
    \label{def:initial_terminal_object}
    Let~\CatC be a category and let~$\Obja \setin \CatC$ be an object.
    We say that~$\Obja$ is an \emph{\iindex{initial object}} if, for all~$\Objb \setin\CatC$, the hom-set~$\HomSet{\CatC}{\Obja}{\Objb}$ has exactly one element.
    We say that~$\Obja$ is a \emph{\iindex{terminal object}} if, for all~$\Objc\setin\CatC$, the hom-set~$\HomSet{\CatC}{\Objc}{\Obja}$ has exactly one element.
\end{ctdefinition}
