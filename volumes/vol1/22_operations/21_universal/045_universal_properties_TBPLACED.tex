\begin{ctdefinition}[Initial and terminal object]
  Let~\CatC be a category and let~$A \in \CatC$ be an object. We say that~$A$ is an \emph{\iindex{initial object}} if, for all~$B \in\CatC$, the hom-set~$\Hom_\CatC(A,B)$ has exactly one element. We say that~$A$ is a \emph{\iindex{terminal object}} if, for all~$D\in\CatC$, the hom-set~$\Hom_\CatC(D,A)$ has exactly one element.
\end{ctdefinition}
