\begin{definition}[Sum of resources]
  \label{def:sum-resources}
  If the poset~$\posgenA$ is monoidal, then the \emph{sum} of~$n$ copies of $\posgenA$ is a design problem given by
  \begin{align*}
    \sum \colon (\posgenA^n)\op \times \posgenA &\toinPos \Bool \\
    \tup{ \F{a_1}^*, \dots, \F{a_n}^*, \R{a_\text{resource}}} &\mapsto (\F{a_1} + \dots + \F{a_n} \posleq_\posgenA \R{a_\text{resource}}).
  \end{align*}

\end{definition}
