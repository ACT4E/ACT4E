\section{Categories \Lat and \BoundedLat}
\begin{ctdefinition}[Category \Lat]
    \SYNDEF{category of lattices and lattices homomorphisms}
    \label{def:Lat}
    The category~\Lat is defined by:
    \begin{enumerate}
        \item \emph{Objects}: The objects of this category are all \SY{lattices}.
        \item \emph{Morphisms}: The morphisms from a \SY{lattice}~$\Obja$ to a \SY{lattice}~$\Objb$ are the \SY{lattice homomorphisms} from~$\Obja$ to~$\Objb$.
        \item \emph{Identity morphism}: The \SY{identity morphism} for the \SY{lattice}~$\Obja$
              is the identity map~$\catidat\Obja$.
        \item \emph{Composition operation}: The composition operation is composition of maps.
    \end{enumerate}
\end{ctdefinition}

\begin{ctdefinition}[Category \BoundedLat]
    \SYNDEF{category of bounded lattices and lattices homomorphisms}
    \label{def:BoundedLat}
    The category~\BoundedLat is defined by:
    \begin{enumerate}
        \item \emph{Objects}: The objects of this category are all \SY{bounded lattices}.
        \item \emph{Morphisms}: The morphisms from a \SY{lattice}~$\Obja$ to a \SY{lattice}~$\Objb$ are the \SY{lattice homomorphisms} from~$\Obja$ to~$\Objb$.
        \item \emph{Identity morphism}: The \SY{identity morphism} for the \SY{bounded lattice}~$\Obja$
              is the identity map~$\catidat\Obja$.
        \item \emph{Composition operation}: The composition operation is composition of maps.
    \end{enumerate}
\end{ctdefinition}

\begin{exercise}
    \label{ex:lat_is_cat}
    Show that \Lat is a category.
\end{exercise}
\begin{solution}
    Clearly, given any \SY{lattice}~$\Obja$, the identity map~$\catidat\Obja$ is a \SY{lattice homomorphism}, since
    \begin{equation}
        \begin{aligned}
            \catidat\Obja(\Objaela \meetof\Obja \Objaelb) & =\Objaela \meetof\Obja \Objaelb \\
            \catidat\Obja(\Objaela \joinof\Obja \Objaelb) & =\Objaela \joinof\Obja \Objaelb.
        \end{aligned}
    \end{equation}
    This said, the identity map satisfies unitality.
    Now, given \SY{lattice homomorphisms}
    \begin{equation}
        \mora\colon \Obja \mto \Objb
        \qqand
        \morb\colon \Objb\mto \Objc,
    \end{equation}
    their composition is a \SY{lattice homomorphism}, since
    \begin{equation}
        \begin{aligned}
            (\morab)(\Objaela \meetof\Obja \Objaelb)
             & = \mora(\Objaela \meetof\Obja \Objaelb) \mthen \morb \\
             & = \morb(\mora(\Objaela) \meetof\Objb \mora(\Objaelb)) \\
             & =(\morab)(\Objaela) \meetof\Objc (\morab)(\Objaelb),
        \end{aligned}
    \end{equation}
    and
    \begin{equation}
        \begin{aligned}
            (\morab)(\Objaela \joinof\Obja \Objaelb)
             & = \mora(\Objaela \joinof\Obja \Objaelb) \mthen \morb \\
             & = \morb(\mora(\Objaela) \joinof\Objb \mora(\Objaelb)) \\
             & =(\morab)(\Objaela) \joinof\Objc (\morab)(\Objaelb),
        \end{aligned}
    \end{equation}
    We have already checked in the past the map composition is \SY{associative} (\eg, when checking that \Set and \Pos are categories).
\end{solution}

\begin{exercise}
    \label{ex:boundlat_is_cat}
    Show that \BoundedLat is a category.
\end{exercise}

\begin{solution}
    Consider the solution of \cref{ex:lat_is_cat} as a starting point.
    Clearly, given any \SY{lattice}~$\Obja$, the identity map~$\catidat\Obja$ is also a bounded \SY{lattice homomorphism}, since
    \begin{equation}
        \begin{aligned}
            \catidat\Obja(\bot_\Obja) & =\bot_\Obja \\
            \catidat\Obja(\top_\Obja) & =\top_\Obja.
        \end{aligned}
    \end{equation}
    This said, the identity map satisfies unitality.
    Now, given \SY{lattice homomorphisms} $\mora\colon \Obja \mto \Objb$ and $\morb\colon \Objb\mto \Objc$, their composition is also a bounded \SY{lattice homomorphism}, since
    \begin{equation}
        \begin{aligned}
            (\morab)(\bot_\Obja)
             & = \morb(\bot_\Objb) \\
             & = \bot_\Objc,
        \end{aligned}
    \end{equation}
    and
    \begin{equation}
        \begin{aligned}
            (\morab)(\top_\Obja)
             & = \morb(\top_\Objb) \\
             & = \top_\Objc.
        \end{aligned}
    \end{equation}
    We have already checked in the past the map composition is \SY{associative} (\eg, when checking that \Set and \Pos are categories).
\end{solution}
