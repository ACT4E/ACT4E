\begin{ctdefinition}[Embedding functor]
    \label{def:embedding-functor}
    \SYNDEF{embedding functor}

    A \SY{functor}~$\funa\colon \CatC \fto \CatD$ is an \emph{embedding} if it is faithful and~$\funaob$ is \SY{injective}.
\end{ctdefinition}
